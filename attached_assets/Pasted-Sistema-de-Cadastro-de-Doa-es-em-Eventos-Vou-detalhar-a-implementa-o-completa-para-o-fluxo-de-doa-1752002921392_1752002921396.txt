Sistema de Cadastro de Doações em Eventos
Vou detalhar a implementação completa para o fluxo de doações durante eventos como a "Semana do Produtor", incluindo:

Pré-cadastro de Eventos e Insumos (pelo Dashboard)

Interface para Técnicos registrarem doações (mobile/tablet/notebook)

Atualização automática dos cards de estatísticas

// components/admin/EventosManager.tsx
import { useState, useEffect } from 'react';
import { db } from '@/lib/firebase';
import { collection, addDoc, doc, updateDoc, deleteDoc, onSnapshot, Timestamp } from 'firebase/firestore';
import { DatePicker } from '@/components/ui/date-picker';

export const EventosManager = () => {
  const [eventos, setEventos] = useState<any[]>([]);
  const [novoEvento, setNovoEvento] = useState({
    nome: '',
    dataInicio: null as Date | null,
    dataFim: null as Date | null,
    ativo: true
  });

  useEffect(() => {
    const unsubscribe = onSnapshot(collection(db, 'eventos'), (snapshot) => {
      setEventos(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
    });
    return () => unsubscribe();
  }, []);

  const handleAddEvento = async () => {
    if (!novoEvento.nome || !novoEvento.dataInicio || !novoEvento.dataFim) return;
    
    try {
      await addDoc(collection(db, 'eventos'), {
        ...novoEvento,
        dataInicio: Timestamp.fromDate(novoEvento.dataInicio),
        dataFim: Timestamp.fromDate(novoEvento.dataFim),
        criadoEm: Timestamp.now()
      });
      setNovoEvento({ nome: '', dataInicio: null, dataFim: null, ativo: true });
    } catch (error) {
      console.error('Erro ao adicionar evento:', error);
    }
  };

  return (
    <div className="p-6">
      <h2 className="text-2xl font-bold mb-6">Gerenciar Eventos</h2>
      
      <div className="bg-white p-6 rounded-lg shadow-md mb-6">
        <h3 className="text-xl font-semibold mb-4">Adicionar Novo Evento</h3>
        
        <div className="grid grid-cols-1 md:grid-cols-3 gap-4 mb-4">
          <div>
            <label className="block text-sm font-medium mb-1">Nome do Evento</label>
            <input
              type="text"
              className="w-full p-2 border rounded"
              value={novoEvento.nome}
              onChange={(e) => setNovoEvento({...novoEvento, nome: e.target.value})}
              placeholder="Semana do Produtor 2024"
            />
          </div>
          
          <div>
            <label className="block text-sm font-medium mb-1">Data Início</label>
            <DatePicker
              selected={novoEvento.dataInicio}
              onChange={(date: Date) => setNovoEvento({...novoEvento, dataInicio: date})}
              className="w-full p-2 border rounded"
            />
          </div>
          
          <div>
            <label className="block text-sm font-medium mb-1">Data Fim</label>
            <DatePicker
              selected={novoEvento.dataFim}
              onChange={(date: Date) => setNovoEvento({...novoEvento, dataFim: date})}
              className="w-full p-2 border rounded"
            />
          </div>
        </div>
        
        <button
          onClick={handleAddEvento}
          className="bg-blue-600 text-white px-4 py-2 rounded hover:bg-blue-700"
          disabled={!novoEvento.nome || !novoEvento.dataInicio || !novoEvento.dataFim}
        >
          Adicionar Evento
        </button>
      </div>
      
      <div className="bg-white p-6 rounded-lg shadow-md">
        <h3 className="text-xl font-semibold mb-4">Eventos Cadastrados</h3>
        
        <div className="overflow-x-auto">
          <table className="min-w-full divide-y divide-gray-200">
            <thead className="bg-gray-50">
              <tr>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Nome</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Período</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Status</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Ações</th>
              </tr>
            </thead>
            <tbody className="bg-white divide-y divide-gray-200">
              {eventos.map((evento) => (
                <tr key={evento.id}>
                  <td className="px-6 py-4 whitespace-nowrap">{evento.nome}</td>
                  <td className="px-6 py-4 whitespace-nowrap">
                    {evento.dataInicio?.toDate().toLocaleDateString()} a {evento.dataFim?.toDate().toLocaleDateString()}
                  </td>
                  <td className="px-6 py-4 whitespace-nowrap">
                    <span className={`px-2 py-1 rounded-full text-xs ${evento.ativo ? 'bg-green-100 text-green-800' : 'bg-gray-100 text-gray-800'}`}>
                      {evento.ativo ? 'Ativo' : 'Inativo'}
                    </span>
                  </td>
                  <td className="px-6 py-4 whitespace-nowrap">
                    <button
                      onClick={() => toggleActive(evento.id, evento.ativo)}
                      className={`mr-2 px-2 py-1 rounded text-xs ${evento.ativo ? 'bg-yellow-500' : 'bg-green-500'} text-white`}
                    >
                      {evento.ativo ? 'Desativar' : 'Ativar'}
                    </button>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </div>
    </div>
  );
};

3. Interface para Técnicos Registrar Doações
typescript
// pages/eventos/[eventoId]/registrar-doacao.tsx
import { useState, useEffect } from 'react';
import { useRouter } from 'next/router';
import { db } from '@/lib/firebase';
import { collection, query, where, getDocs, addDoc } from 'firebase/firestore';
import { useAuth } from '@/contexts/AuthContext';

export default function RegistrarDoacao() {
  const router = useRouter();
  const { eventoId } = router.query;
  const { user } = useAuth();
  
  const [evento, setEvento] = useState<any>(null);
  const [insumos, setInsumos] = useState<any[]>([]);
  const [formData, setFormData] = useState({
    insumoId: '',
    quantidade: '',
    beneficiarioNome: '',
    beneficiarioCpf: '',
    beneficiarioPropriedade: ''
  });
  const [loading, setLoading] = useState(true);
  const [success, setSuccess] = useState(false);

  useEffect(() => {
    if (!eventoId) return;

    const fetchData = async () => {
      try {
        // Buscar dados do evento
        const eventoDoc = await getDoc(doc(db, 'eventos', eventoId as string));
        if (!eventoDoc.exists()) {
          router.push('/eventos');
          return;
        }
        setEvento({ id: eventoDoc.id, ...eventoDoc.data() });

        // Buscar insumos ativos
        const q = query(collection(db, 'insumos'), where('ativo', '==', true));
        const querySnapshot = await getDocs(q);
        setInsumos(querySnapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));

        setLoading(false);
      } catch (error) {
        console.error('Erro ao buscar dados:', error);
        setLoading(false);
      }
    };

    fetchData();
  }, [eventoId, router]);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    
    if (!formData.insumoId || !formData.quantidade || !user) return;
    
    try {
      await addDoc(collection(db, 'doacoes_evento'), {
        eventoId,
        insumoId: formData.insumoId,
        quantidade: Number(formData.quantidade),
        tecnico: {
          id: user.uid,
          nome: user.displayName || 'Técnico'
        },
        beneficiario: {
          nome: formData.beneficiarioNome,
          ...(formData.beneficiarioCpf && { cpf: formData.beneficiarioCpf }),
          ...(formData.beneficiarioPropriedade && { propriedade: formData.beneficiarioPropriedade })
        },
        timestamp: new Date()
      });

      setSuccess(true);
      setFormData({
        insumoId: '',
        quantidade: '',
        beneficiarioNome: '',
        beneficiarioCpf: '',
        beneficiarioPropriedade: ''
      });

      // Resetar mensagem de sucesso após 3 segundos
      setTimeout(() => setSuccess(false), 3000);
    } catch (error) {
      console.error('Erro ao registrar doação:', error);
    }
  };

  if (loading) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="animate-spin rounded-full h-12 w-12 border-t-2 border-b-2 border-blue-500"></div>
      </div>
    );
  }

  if (!evento) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <p className="text-red-500">Evento não encontrado</p>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 py-12 px-4 sm:px-6 lg:px-8">
      <div className="max-w-md mx-auto bg-white rounded-xl shadow-md overflow-hidden md:max-w-2xl">
        <div className="p-8">
          <div className="text-center mb-6">
            <h1 className="text-2xl font-bold text-gray-900">{evento.nome}</h1>
            <p className="mt-2 text-gray-600">Registrar nova doação</p>
          </div>
          
          {success && (
            <div className="mb-4 p-4 bg-green-100 text-green-700 rounded-lg">
              Doação registrada com sucesso!
            </div>
          )}
          
          <form onSubmit={handleSubmit} className="space-y-6">
            <div>
              <label htmlFor="insumoId" className="block text-sm font-medium text-gray-700">
                Insumo Doado
              </label>
              <select
                id="insumoId"
                name="insumoId"
                value={formData.insumoId}
                onChange={(e) => setFormData({...formData, insumoId: e.target.value})}
                className="mt-1 block w-full pl-3 pr-10 py-2 text-base border-gray-300 focus:outline-none focus:ring-blue-500 focus:border-blue-500 sm:text-sm rounded-md"
                required
              >
                <option value="">Selecione um insumo</option>
                {insumos.map((insumo) => (
                  <option key={insumo.id} value={insumo.id}>
                    {insumo.nome} ({insumo.unidade})
                  </option>
                ))}
              </select>
            </div>
            
            <div>
              <label htmlFor="quantidade" className="block text-sm font-medium text-gray-700">
                Quantidade
              </label>
              <input
                type="number"
                id="quantidade"
                name="quantidade"
                value={formData.quantidade}
                onChange={(e) => setFormData({...formData, quantidade: e.target.value})}
                className="mt-1 block w-full shadow-sm sm:text-sm focus:ring-blue-500 focus:border-blue-500 border-gray-300 rounded-md"
                placeholder="Ex: 100"
                min="1"
                required
              />
            </div>
            
            <div className="border-t border-gray-200 pt-4">
              <h3 className="text-lg font-medium text-gray-900 mb-4">Dados do Beneficiário</h3>
              
              <div className="space-y-4">
                <div>
                  <label htmlFor="beneficiarioNome" className="block text-sm font-medium text-gray-700">
                    Nome Completo
                  </label>
                  <input
                    type="text"
                    id="beneficiarioNome"
                    name="beneficiarioNome"
                    value={formData.beneficiarioNome}
                    onChange={(e) => setFormData({...formData, beneficiarioNome: e.target.value})}
                    className="mt-1 block w-full shadow-sm sm:text-sm focus:ring-blue-500 focus:border-blue-500 border-gray-300 rounded-md"
                    required
                  />
                </div>
                
                <div>
                  <label htmlFor="beneficiarioCpf" className="block text-sm font-medium text-gray-700">
                    CPF (opcional)
                  </label>
                  <input
                    type="text"
                    id="beneficiarioCpf"
                    name="beneficiarioCpf"
                    value={formData.beneficiarioCpf}
                    onChange={(e) => setFormData({...formData, beneficiarioCpf: e.target.value})}
                    className="mt-1 block w-full shadow-sm sm:text-sm focus:ring-blue-500 focus:border-blue-500 border-gray-300 rounded-md"
                  />
                </div>
                
                <div>
                  <label htmlFor="beneficiarioPropriedade" className="block text-sm font-medium text-gray-700">
                    Nome da Propriedade (opcional)
                  </label>
                  <input
                    type="text"
                    id="beneficiarioPropriedade"
                    name="beneficiarioPropriedade"
                    value={formData.beneficiarioPropriedade}
                    onChange={(e) => setFormData({...formData, beneficiarioPropriedade: e.target.value})}
                    className="mt-1 block w-full shadow-sm sm:text-sm focus:ring-blue-500 focus:border-blue-500 border-gray-300 rounded-md"
                  />
                </div>
              </div>
            </div>
            
            <div>
              <button
                type="submit"
                className="w-full flex justify-center py-2 px-4 border border-transparent rounded-md shadow-sm text-sm font-medium text-white bg-blue-600 hover:bg-blue-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-blue-500"
              >
                Registrar Doação
              </button>
            </div>
          </form>
        </div>
      </div>
    </div>
  );
}

// components/admin/EventStatsConfig.tsx
import { useState, useEffect } from 'react';
import { db } from '@/lib/firebase';
import { collection, addDoc, doc, updateDoc, deleteDoc, onSnapshot } from 'firebase/firestore';

export const EventStatsConfig = ({ eventoId }: { eventoId: string }) => {
  const [configs, setConfigs] = useState<any[]>([]);
  const [insumos, setInsumos] = useState<any[]>([]);
  const [novaConfig, setNovaConfig] = useState({
    titulo: '',
    insumoId: '',
    tipoAgregacao: 'sum' as 'sum' | 'count',
    periodo: 'hoje' as 'hoje' | 'evento',
    ativo: true
  });

  useEffect(() => {
    // Buscar configurações existentes para este evento
    const unsubscribeConfigs = onSnapshot(
      query(collection(db, 'estatisticas_evento'), where('eventoId', '==', eventoId)),
      (snapshot) => {
        setConfigs(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
      }
    );

    // Buscar insumos ativos
    const unsubscribeInsumos = onSnapshot(
      query(collection(db, 'insumos'), where('ativo', '==', true)),
      (snapshot) => {
        setInsumos(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
      }
    );

    return () => {
      unsubscribeConfigs();
      unsubscribeInsumos();
    };
  }, [eventoId]);

  const handleAddConfig = async () => {
    if (!novaConfig.titulo || !novaConfig.insumoId) return;
    
    try {
      await addDoc(collection(db, 'estatisticas_evento'), {
        ...novaConfig,
        eventoId,
        criadoEm: new Date()
      });
      setNovaConfig({
        titulo: '',
        insumoId: '',
        tipoAgregacao: 'sum',
        periodo: 'hoje',
        ativo: true
      });
    } catch (error) {
      console.error('Erro ao adicionar configuração:', error);
    }
  };

  return (
    <div className="p-6">
      <h2 className="text-2xl font-bold mb-6">Configurar Estatísticas do Evento</h2>

        <div className="grid grid-cols-1 md:grid-cols-2 gap-4 mb-4">
          <div>
            <label className="block text-sm font-medium mb-1">Título</label>
            <input
              type="text"
              className="w-full p-2 border rounded"
              value={novaConfig.titulo}
              onChange={(e) => setNovaConfig({...novaConfig, titulo: e.target.value})}
              placeholder="Ex: Pintos doados hoje"
            />
          </div>
          
          <div>
            <label className="block text-sm font-medium mb-1">Insumo</label>
            <select
              className="w-full p-2 border rounded"
              value={novaConfig.insumoId}
              onChange={(e) => setNovaConfig({...novaConfig, insumoId: e.target.value})}
            >
              <option value="">Selecione um insumo</option>
              {insumos.map((insumo) => (
                <option key={insumo.id} value={insumo.id}>
                  {insumo.nome} ({insumo.unidade})
                </option>
              ))}
            </select>
          </div>
          
          <div>
            <label className="block text-sm font-medium mb-1">Tipo de Agregação</label>
            <select
              className="w-full p-2 border rounded"
              value={novaConfig.tipoAgregacao}
              onChange={(e) => setNovaConfig({...novaConfig, tipoAgregacao: e.target.value as any})}
            >
              <option value="sum">Soma total</option>

typescript
// pages/eventos/[eventoId]/telao.tsx
import { useEffect, useState } from 'react';
import { useRouter } from 'next/router';
import { db } from '@/lib/firebase';
import { collection, query, where, onSnapshot } from 'firebase/firestore';
import { DynamicStatCard } from '@/components/DynamicStatCard';

export default function EventoTelao() {
  const router = useRouter();
  const { eventoId } = router.query;
  
  const [evento, setEvento] = useState<any>(null);
  const [statsConfig, setStatsConfig] = useState<any[]>([]);
  const [insumos, setInsumos] = useState<any[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    if (!eventoId) return;

    const unsubscribe = onSnapshot(
      query(collection(db, 'estatisticas_evento'), 
      where('eventoId', '==', eventoId),
      where('ativo', '==', true),
      (snapshot) => {
        setStatsConfig(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
      }
    );

    // Buscar dados do evento e insumos uma vez
    const fetchData = async () => {
      try {
        const [eventoDoc, insumosSnapshot] = await Promise.all([
          getDoc(doc(db, 'eventos', eventoId as string)),
          getDocs(collection(db, 'insumos'))
        ]);

        if (eventoDoc.exists()) {
          setEvento({ id: eventoDoc.id, ...eventoDoc.data() });
        }

        setInsumos(insumosSnapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
        setLoading(false);
      } catch (error) {
        console.error('Erro ao buscar dados:', error);
        setLoading(false);
      }
    };

    fetchData();

    return () => unsubscribe();
  }, [eventoId]);

  if (loading) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-gray-900">
        <div className="animate-spin rounded-full h-12 w-12 border-t-2 border-b-2 border-blue-500"></div>
      </div>
    );
  }

  if (!evento) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-gray-900 text-white">
        <p>Evento não encontrado</p>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-900 p-8">
      <div className="text-center mb-12">
        <h1 className="text-4xl font-bold text-white mb-2">{evento.nome}</h1>
        <p className="text-xl text-gray-300">Estatísticas em tempo real</p>
      </div>
      
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
        {statsConfig.map((config) => {
          const insumo = insumos.find(i => i.id === config.insumoId);
          
          // Criar configuração para o DynamicStatCard
          const cardConfig = {
            id: config.id,
            titulo: config.titulo,
            colecaoFonte: 'doacoes_evento',
            campo: config.tipoAgregacao === 'sum' ? 'quantidade' : '', // para count não precisamos de campo
            tipoAgregacao: config.tipoAgregacao,
            periodo: config.periodo === 'hoje' ? 'hoje' : 'todos',
            unidade: insumo?.unidade || '',
            filtroAdicional: [
              where('eventoId', '==', eventoId),
              ...(config.insumoId ? [where('insumoId', '==', config.insumoId)] : [])
            ]
          };
          
          return (
            <div key={config.id} className="transform hover:scale-105 transition-transform">
              <DynamicStatCard 
                config={cardConfig}
                variant="transparent"
              />
            </div>
          );
        })}
      </div>
      
      <div className="mt-12 text-center text-gray-400">
        <p>Atualizado em tempo real - {new Date().toLocaleTimeString()}</p>
      </div>
    </div>
  );
}
