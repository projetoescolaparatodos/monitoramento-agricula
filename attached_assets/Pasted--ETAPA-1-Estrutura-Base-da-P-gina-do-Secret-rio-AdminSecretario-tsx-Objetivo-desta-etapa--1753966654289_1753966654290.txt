✅ ETAPA 1 — Estrutura Base da Página do Secretário (AdminSecretario.tsx)

📌 Objetivo desta etapa: Criar a estrutura inicial com abas para cada funcionalidade principal.
🧠 Planejamento

A página do secretário será uma interface centralizada com visualização dos dados de vários setores, principalmente a Pesca.

Vamos criar abas (tabs) ou um menu lateral para organizar as seguintes seções:
Aba	Componente sugerido (nome)	Origem dos dados
Solicitações de Cadastro	SolicitacoesCadastro.tsx	Firebase (coleção: solicitacoes)
Gráfico de Áreas Atendidas	GraficoAtendimentos.tsx	Firebase ou função agregadora
Doações	PainelDoacoes.tsx	Firebase (coleção: doacoes)
Visitas Técnicas (Pesca)	RelatorioVisitasTecnicas.tsx	Firebase (coleção: visitas_tecnicas)
Viveiros em Construção	MetadadosViveiros.tsx	Firebase (coleção: viveiros_construcao)
✅ 1. Criar o arquivo AdminSecretario.tsx com abas

// src/pages/AdminSecretario.tsx

import React, { useState } from 'react';
import SolicitacoesCadastro from '@/components/secretario/SolicitacoesCadastro';
import GraficoAtendimentos from '@/components/secretario/GraficoAtendimentos';
import PainelDoacoes from '@/components/secretario/PainelDoacoes';
import RelatorioVisitasTecnicas from '@/components/secretario/RelatorioVisitasTecnicas';
import MetadadosViveiros from '@/components/secretario/MetadadosViveiros';

const abas = [
  'Solicitações de Cadastro',
  'Áreas Atendidas',
  'Doações',
  'Visitas Técnicas (Pesca)',
  'Viveiros em Construção',
];

export default function AdminSecretario() {
  const [abaAtiva, setAbaAtiva] = useState(0);

  return (
    <div className="p-4">
      <h1 className="text-2xl font-bold mb-4">Painel do Secretário</h1>
      
      <div className="flex space-x-2 border-b mb-4">
        {abas.map((aba, idx) => (
          <button
            key={aba}
            className={`px-4 py-2 ${
              idx === abaAtiva
                ? 'border-b-2 border-blue-500 font-semibold'
                : 'text-gray-500'
            }`}
            onClick={() => setAbaAtiva(idx)}
          >
            {aba}
          </button>
        ))}
      </div>

      <div className="mt-4">
        {abaAtiva === 0 && <SolicitacoesCadastro />}
        {abaAtiva === 1 && <GraficoAtendimentos />}
        {abaAtiva === 2 && <PainelDoacoes />}
        {abaAtiva === 3 && <RelatorioVisitasTecnicas />}
        {abaAtiva === 4 && <MetadadosViveiros />}
      </div>
    </div>
  );
}
