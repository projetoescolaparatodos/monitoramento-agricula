Solução para o Loop Infinito
1. Atualize o FormAgricultura (index.tsx)
tsx
import React, { useState, useEffect, useRef } from 'react';
import { useNavigate } from 'react-router-dom';
import Card from '../../components/ui/card';
import ErrorBoundary from '../../components/common/ErrorBoundary';

const FormAgricultura = () => {
  const isMountedRef = useRef(true);
  const navigate = useNavigate();
  const [isSubmitting, setIsSubmitting] = useState(false);

  // Efeito de limpeza seguro
  useEffect(() => {
    return () => {
      isMountedRef.current = false;
      console.log('🧹 Limpeza segura do FormAgricultura executada');
    };
  }, []);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!isMountedRef.current || isSubmitting) return;

    setIsSubmitting(true);
    try {
      // Sua lógica de submit aqui
      console.log('✅ Formulário enviado com sucesso');
      navigate('/sucesso'); // Redirecionar após sucesso
    } catch (error) {
      console.error('Erro no submit:', error);
    } finally {
      if (isMountedRef.current) {
        setIsSubmitting(false);
      }
    }
  };

  return (
    <ErrorBoundary fallback={<div>Erro no formulário</div>}>
      <Card key={Date.now()}> {/* Chave única para evitar reutilização */}
        <form onSubmit={handleSubmit}>
          {/* Seus campos de formulário aqui */}
          
          <button 
            type="submit" 
            disabled={isSubmitting}
            className="bg-green-600 hover:bg-green-700 text-white py-2 px-4 rounded disabled:opacity-50"
          >
            {isSubmitting ? 'Processando...' : 'Cadastrar'}
          </button>
        </form>
      </Card>
    </ErrorBoundary>
  );
};

export default FormAgricultura;
