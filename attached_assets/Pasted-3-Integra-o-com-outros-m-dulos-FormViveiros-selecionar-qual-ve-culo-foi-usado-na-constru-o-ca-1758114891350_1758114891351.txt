6. Exemplo de Componentes
CadastrarVeiculoForm.tsx
import { useState } from "react";
import { addDoc, collection } from "firebase/firestore";
import { db } from "@/lib/firebase";

export default function CadastrarVeiculoForm() {
  const [modelo, setModelo] = useState("");
  const [tipo, setTipo] = useState("caminhonete");
  const [consumo, setConsumo] = useState("");
  const [status, setStatus] = useState("funcionando");

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    await addDoc(collection(db, "veiculos"), {
      modelo,
      tipo,
      consumoMedio: parseFloat(consumo),
      status,
      manutencoes: []
    });
    setModelo("");
    setConsumo("");
  };

  return (
    <form onSubmit={handleSubmit} className="space-y-4 p-4 border rounded-lg">
      <input
        value={modelo}
        onChange={(e) => setModelo(e.target.value)}
        placeholder="Modelo do veículo"
        className="border p-2 rounded w-full"
        required
      />
      <select value={tipo} onChange={(e) => setTipo(e.target.value)} className="border p-2 rounded w-full">
        <option value="caminhonete">Caminhonete</option>
        <option value="carro">Carro</option>
        <option value="trator">Trator</option>
      </select>
      <input
        type="number"
        step="0.1"
        value={consumo}
        onChange={(e) => setConsumo(e.target.value)}
        placeholder="Consumo médio (km/L)"
        className="border p-2 rounded w-full"
        required
      />
      <select value={status} onChange={(e) => setStatus(e.target.value)} className="border p-2 rounded w-full">
        <option value="funcionando">Funcionando</option>
        <option value="quebrado">Quebrado</option>
      </select>
      <button className="bg-green-600 text-white px-4 py-2 rounded">Cadastrar</button>
    </form>
  );
}
