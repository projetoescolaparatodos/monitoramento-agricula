2. Componente DynamicStatisticCard
Vamos criar um novo componente para estatísticas dinâmicas:

typescript
// components/DynamicStatisticCard.tsx
import { useEffect, useState } from 'react';
import { db } from '@/lib/firebase';
import { collection, query, where, onSnapshot } from 'firebase/firestore';
import { Card, CardContent } from '@/components/ui/card';

interface DynamicStatisticCardProps {
  config: {
    id: string;
    titulo: string;
    colecaoFonte: string;
    campo: string;
    tipoAgregacao: 'sum' | 'avg' | 'max' | 'count';
    periodo: string;
    unidade: string;
  };
  variant?: 'default' | 'transparent';
}

export const DynamicStatisticCard = ({ config, variant = 'default' }: DynamicStatisticCardProps) => {
  const [value, setValue] = useState<number>(0);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const calcularPeriodo = () => {
      const now = new Date();
      let startDate = new Date();
      
      switch(config.periodo) {
        case 'hoje':
          startDate.setHours(0, 0, 0, 0);
          break;
        case '30dias':
          startDate.setDate(now.getDate() - 30);
          break;
        case 'mesAtual':
          startDate = new Date(now.getFullYear(), now.getMonth(), 1);
          break;
        case 'safraAtual':
          // Ajustar conforme lógica da safra
          startDate = new Date(now.getFullYear(), 6, 1); // Ex: 1 de julho
          break;
        default:
          startDate = new Date(0); // Todos os dados
      }
      
      return { startDate, endDate: now };
    };

    const { startDate, endDate } = calcularPeriodo();
    
    const q = query(
      collection(db, config.colecaoFonte),
      where('timestamp', '>=', startDate),
      where('timestamp', '<=', endDate)
    );

    const unsubscribe = onSnapshot(q, (snapshot) => {
      let calculatedValue = 0;
      
      if (config.tipoAgregacao === 'count') {
        calculatedValue = snapshot.size;
      } else {
        const values = snapshot.docs.map(doc => doc.data()[config.campo] || 0);
        
        switch(config.tipoAgregacao) {
          case 'sum':
            calculatedValue = values.reduce((a, b) => a + b, 0);
            break;
          case 'avg':
            calculatedValue = values.reduce((a, b) => a + b, 0) / (values.length || 1);
            break;
          case 'max':
            calculatedValue = Math.max(...values);
            break;
        }
      }
      
      setValue(calculatedValue);
      setLoading(false);
    });

    return () => unsubscribe();
  }, [config]);

  return (
    <Card className={`${variant === "transparent" ? "bg-white/90 backdrop-blur-sm" : "bg-white"} rounded-lg shadow-lg hover:shadow-xl transition-all duration-300 overflow-hidden transform hover:-translate-y-1 relative`}>
      <div className="absolute top-2 right-2 bg-green-500 text-white text-xs px-2 py-1 rounded-full">
        Dinâmico
      </div>
      <div className="border-t-4 border-[#00ff4c]"></div>
      <CardContent className="p-6 text-center">
        {loading ? (
          <div className="animate-pulse">
            <div className="h-10 bg-gray-200 rounded mb-3 mx-auto w-3/4"></div>
            <div className="h-4 bg-gray-200 rounded mb-2 mx-auto w-1/2"></div>
          </div>
        ) : (
          <>
            <div className="text-4xl font-bold text-primary-dark mb-3">
              {value.toLocaleString()}
            </div>
            <div className="text-sm uppercase font-semibold tracking-wider mb-3 text-gray-700">
              {config.titulo} {config.unidade && `(${config.unidade})`}
            </div>
          </>
        )}
      </CardContent>
    </Card>
  );
};
3. Atualização do StatisticsSection
Agora vamos modificar o componente existente para incluir as estatísticas dinâmicas:

typescript
// components/StatisticsSection.tsx
import { useQuery } from "@tanstack/react-query";
import { StatisticItem } from "@/types";
import { Card, CardContent } from "@/components/ui/card";
import { ChevronUp, ChevronDown } from "lucide-react";
import { DynamicStatisticCard } from "./DynamicStatisticCard";

const StatisticsSection = ({ variant = "default" }: { variant?: "default" | "transparent" }) => {
  // Query para estatísticas fixas
  const { data: statistics, isLoading: isLoadingFixed } = useQuery<StatisticItem[]>({
    queryKey: ['/api/statistics'],
  });

  // Query para configurações de estatísticas dinâmicas
  const { data: dynamicStatsConfig, isLoading: isLoadingDynamic } = useQuery({
    queryKey: ['dynamic-statistics-config'],
    queryFn: async () => {
      // Implementar fetch das configurações do Firebase
      // Exemplo simplificado:
      const response = await fetch('/api/dynamic-stats-config');
      return response.json();
    }
  });

  // Filtrar estatísticas fixas ativas
  const activeStatistics = statistics?.filter(stat => stat.active !== false && stat.pageType === 'home')
    .sort((a, b) => (a.order || 0) - (b.order || 0));

  return (
    <section id="estatisticas" className="mb-16">
      <div className="text-center mb-10">
        <h2 className={`text-3xl font-bold mb-3 ${variant === "transparent" ? "text-white" : "text-primary"}`}>
          Estatísticas Principais
        </h2>
        <p className={`text-xl md:text-2xl font-medium tracking-wide max-w-3xl mx-auto ${variant === "transparent" ? "text-white" : "text-gray-600"}`}>
          Dados Sobre a produção em Vitória do Xingu
        </p>
      </div>

      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
        {/* Loading state */}
        {(isLoadingFixed || isLoadingDynamic) ? (
          Array(4).fill(0).map((_, index) => (
            <div key={index} className={`${variant === "transparent" ? "bg-white/80" : "bg-white"} rounded-lg shadow-lg p-6 text-center border-t-4 border-primary animate-pulse`}>
              <div className="h-10 bg-gray-200 rounded mb-2 mx-auto w-3/4"></div>
              <div className="h-4 bg-gray-200 rounded mb-2 mx-auto w-1/2"></div>
              <div className="h-4 bg-gray-200 rounded mx-auto w-2/3"></div>
            </div>
          ))
        ) : (
          <>
            {/* Estatísticas fixas */}
            {activeStatistics?.map((stat) => (
              <Card key={stat.id} className={`${variant === "transparent" ? "bg-white/90 backdrop-blur-sm" : "bg-white"} rounded-lg shadow-lg hover:shadow-xl transition-all duration-300 overflow-hidden transform hover:-translate-y-1`}>
                <div className="border-t-4 border-[#00ff4c]"></div>
                <CardContent className="p-6 text-center">
                  <div className="text-4xl font-bold text-primary-dark mb-3">{stat.value}</div>
                  <div className="text-sm uppercase font-semibold tracking-wider mb-3 text-gray-700">{stat.label}</div>
                  {stat.trend && (
                    <div className="flex items-center justify-center text-sm font-medium">
                      <span className={`flex items-center ${stat.isPositive ? "text-green-600" : "text-red-600"}`}>
                        {stat.isPositive ? <ChevronUp className="w-5 h-5 mr-1" /> : <ChevronDown className="w-5 h-5 mr-1" />}
                        {stat.trendValue && <span className="mr-1">{stat.trendValue}</span>} {stat.trend}
                      </span>
                    </div>
                  )}
                </CardContent>
              </Card>
            ))}

            {/* Estatísticas dinâmicas */}
            {dynamicStatsConfig?.filter((config: any) => config.ativo).map((config: any) => (
              <DynamicStatisticCard key={config.id} config={config} variant={variant} />
            ))}
          </>
        )}
      </div>
    </section>
  );
};

export default StatisticsSection;
4. API Routes para Configurações
Precisamos criar uma API route para buscar as configurações:

typescript
// pages/api/dynamic-stats-config.ts
import { db } from '@/lib/firebase';
import { collection, getDocs } from 'firebase/firestore';
import { NextApiRequest, NextApiResponse } from 'next';

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  try {
    const querySnapshot = await getDocs(collection(db, 'estatisticas_dinamicas'));
    const configs = querySnapshot.docs.map(doc => ({
      id: doc.id,
      ...doc.data(),
      // Convert Firestore Timestamp to ISO string if needed
      createdAt: doc.data().createdAt?.toDate().toISOString()
    }));
    
    res.status(200).json(configs);
  } catch (error) {
    console.error('Error fetching dynamic stats config:', error);
    res.status(500).json({ error: 'Failed to fetch dynamic stats configuration' });
  }
}
5. Dashboard de Administração
Vamos criar um painel para gerenciar as estatísticas dinâmicas:

typescript
// components/admin/DynamicStatsManager.tsx
import { useState, useEffect } from 'react';
import { db } from '@/lib/firebase';
import { collection, addDoc, doc, updateDoc, deleteDoc, onSnapshot } from 'firebase/firestore';

export const DynamicStatsManager = () => {
  const [configs, setConfigs] = useState<any[]>([]);
  const [newConfig, setNewConfig] = useState({
    titulo: '',
    colecaoFonte: 'mapa_atividades',
    campo: '',
    tipoAgregacao: 'sum',
    periodo: '30dias',
    unidade: '',
    ativo: true,
    ordem: 0
  });

  useEffect(() => {
    const unsubscribe = onSnapshot(collection(db, 'estatisticas_dinamicas'), (snapshot) => {
      setConfigs(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
    });
    
    return () => unsubscribe();
  }, []);

  const handleAddConfig = async () => {
    try {
      await addDoc(collection(db, 'estatisticas_dinamicas'), {
        ...newConfig,
        createdAt: new Date()
      };
      setNewConfig({
        titulo: '',
        colecaoFonte: 'mapa_atividades',
        campo: '',
        tipoAgregacao: 'sum',
        periodo: '30dias',
        unidade: '',
        ativo: true,
        ordem: 0
      });
    } catch (error) {
      console.error('Error adding config:', error);
    }
  };

  const toggleActive = async (id: string, currentActive: boolean) => {
    try {
      await updateDoc(doc(db, 'estatisticas_dinamicas', id), {
        ativo: !currentActive
      };
    } catch (error) {
      console.error('Error toggling active:', error);
    }
  };

  const deleteConfig = async (id: string) => {
    try {
      await deleteDoc(doc(db, 'estatisticas_dinamicas', id));
    } catch (error) {
      console.error('Error deleting config:', error);
    }
  };

  return (
    <div className="p-6">
      <h2 className="text-2xl font-bold mb-6">Gerenciar Estatísticas Dinâmicas</h2>
      
      <div className="bg-white p-6 rounded-lg shadow-md mb-6">
        <h3 className="text-xl font-semibold mb-4">Adicionar Nova Estatística</h3>
        
        <div className="grid grid-cols-1 md:grid-cols-2 gap-4 mb-4">
          <div>
            <label className="block text-sm font-medium mb-1">Título</label>
            <input
              type="text"
              className="w-full p-2 border rounded"
              value={newConfig.titulo}
              onChange={(e) => setNewConfig({...newConfig, titulo: e.target.value})}
            />
          </div>
          
          <div>
            <label className="block text-sm font-medium mb-1">Fonte de Dados</label>
            <select
              className="w-full p-2 border rounded"
              value={newConfig.colecaoFonte}
              onChange={(e) => setNewConfig({...newConfig, colecaoFonte: e.target.value})}
            >
              <option value="mapa_atividades">Mapa de Atividades</option>
              <option value="doacoes_evento">Doações em Eventos</option>
            </select>
          </div>
          
          <div>
            <label className="block text-sm font-medium mb-1">Campo Numérico</label>
            <input
              type="text"
              className="w-full p-2 border rounded"
              value={newConfig.campo}
              onChange={(e) => setNewConfig({...newConfig, campo: e.target.value})}
              placeholder="horasMaquina, quantidade, etc"
            />
          </div>
          
          <div>
            <label className="block text-sm font-medium mb-1">Tipo de Agregação</label>
            <select
              className="w-full p-2 border rounded"
              value={newConfig.tipoAgregacao}
              onChange={(e) => setNewConfig({...newConfig, tipoAgregacao: e.target.value as any})}
            >
              <option value="sum">Soma</option>
              <option value="avg">Média</option>
              <option value="max">Máximo</option>
              <option value="count">Contagem</option>
            </select>
          </div>
          
          <div>
            <label className="block text-sm font-medium mb-1">Período</label>
            <select
              className="w-full p-2 border rounded"
              value={newConfig.periodo}
              onChange={(e) => setNewConfig({...newConfig, periodo: e.target.value})}
            >
              <option value="hoje">Hoje</option>
              <option value="30dias">Últimos 30 dias</option>
              <option value="mesAtual">Mês atual</option>
              <option value="safraAtual">Safra atual</option>
            </select>
          </div>
          
          <div>
            <label className="block text-sm font-medium mb-1">Unidade (opcional)</label>
            <input
              type="text"
              className="w-full p-2 border rounded"
              value={newConfig.unidade}
              onChange={(e) => setNewConfig({...newConfig, unidade: e.target.value})}
              placeholder="horas, mudas, etc"
            />
          </div>
        </div>
        
        <button
          onClick={handleAddConfig}
          className="bg-green-600 text-white px-4 py-2 rounded hover:bg-green-700"
        >
          Adicionar Estatística
        </button>
      </div>
      
      <div className="bg-white p-6 rounded-lg shadow-md">
        <h3 className="text-xl font-semibold mb-4">Estatísticas Configuradas</h3>
        
        <div className="overflow-x-auto">
          <table className="min-w-full divide-y divide-gray-200">
            <thead className="bg-gray-50">
              <tr>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Título</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Fonte</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Campo</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Agregação</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Período</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Ações</th>
              </tr>
            </thead>
            <tbody className="bg-white divide-y divide-gray-200">
              {configs.map((config) => (
                <tr key={config.id}>
                  <td className="px-6 py-4 whitespace-nowrap">{config.titulo}</td>
                  <td className="px-6 py-4 whitespace-nowrap">{config.colecaoFonte}</td>
                  <td className="px-6 py-4 whitespace-nowrap">{config.campo}</td>
                  <td className="px-6 py-4 whitespace-nowrap">{config.tipoAgregacao}</td>
                  <td className="px-6 py-4 whitespace-nowrap">{config.periodo}</td>
                  <td className="px-6 py-4 whitespace-nowrap">
                    <button
                      onClick={() => toggleActive(config.id, config.ativo)}
                      className={`mr-2 px-2 py-1 rounded text-xs ${config.ativo ? 'bg-yellow-500' : 'bg-green-500'} text-white`}
                    >
                      {config.ativo ? 'Desativar' : 'Ativar'}
                    </button>
                    <button
                      onClick={() => deleteConfig(config.id)}
                      className="bg-red-500 text-white px-2 py-1 rounded text-xs"
                    >
                      Excluir
                    </button>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </div>
    </div>
  );
};

Considerações Finais
Esta implementação oferece:

Estatísticas Dinâmicas em Tempo Real: Atualização automática conforme os dados são modificados no Firebase.

Flexibilidade: O administrador pode criar novas estatísticas sem precisar de alterações de código.

Integração com Mapa Existente: Utiliza os dados já coletados pelo sistema de acompanhamento de atividades.

Suporte a Eventos Especiais: Configurações específicas para telões durante eventos como o Dia do Agricultor.

Experiência Consistente: Mantém o mesmo visual dos cards existentes, com adição do selo "Dinâmico".

Para implementar completamente, será necessário:

Configurar as regras de segurança no Firebase para permitir leitura/escrita apenas para usuários autorizados.

Adicionar o componente DynamicStatsManager ao painel administrativo.

Testar a performance com grandes volumes de dados, considerando a implementação de paginação se necessário.

Adicionar tratamento de erro mais robusto em todos os componentes.
