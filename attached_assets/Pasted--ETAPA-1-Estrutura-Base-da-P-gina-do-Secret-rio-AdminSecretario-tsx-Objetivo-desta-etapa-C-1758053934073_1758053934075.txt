📁 utils/agruparLocalidades.ts
import stringSimilarity from 'string-similarity';

export function agruparLocalidadesPorSimilaridade(dados: { localidade: string }[], limiar = 0.75) {
  const grupos: Record<string, number> = {};

  dados.forEach(({ localidade }) => {
    const nomesExistentes = Object.keys(grupos);
    const match = stringSimilarity.findBestMatch(localidade, nomesExistentes);
    
    if (match.bestMatch.rating > limiar) {
      grupos[match.bestMatch.target]++;
    } else {
      grupos[localidade] = 1;
    }
  });

  return grupos;
}

// src/utils/fetchAtendimentos.ts
import { getDocs, collection } from "firebase/firestore";
import { db } from "@/services/firebase";

export const fetchAtendimentosDeTodosOsSetores = async () => {
  const colecoes = ["agricultura", "pesca", "paa"]; // nomes das coleções de cada mapa

  let atendimentos: any[] = [];

  for (const setor of colecoes) {
    const snapshot = await getDocs(collection(db, setor));
    snapshot.forEach((doc) => {
      atendimentos.push({ ...doc.data(), setor });
    });
  }

  return atendimentos;
};

🔹 Passo 2: Função para normalizar e agrupar localidade

// src/utils/agruparLocalidades.ts
import stringSimilarity from "string-similarity";

export const agruparPorLocalidadeSemelhante = (dados: any[], chave = "localidade") => {
  const grupos: { [key: string]: any[] } = {};

  const nomesPadronizados: string[] = [];

  for (const item of dados) {
    const nomeOriginal = item[chave]?.toLowerCase().replace(/[\W_]+/g, " ").trim();

    if (!nomeOriginal) continue;

    const similar = stringSimilarity.findBestMatch(nomeOriginal, nomesPadronizados);

    if (similar.bestMatch.rating > 0.8) {
      const nomeExistente = similar.bestMatch.target;
      grupos[nomeExistente].push(item);
    } else {
      nomesPadronizados.push(nomeOriginal);
      grupos[nomeOriginal] = [item];
    }
  }

  return grupos;
};

import { collection, getDocs } from 'firebase/firestore';
import { db } from '@/services/firebase';
import { useEffect, useState } from 'react';

const useVisitasTecnicasPesca = () => {
  const [visitas, setVisitas] = useState([]);

  useEffect(() => {
    const fetchData = async () => {
      const colRef = collection(db, 'pesca_visitas_tecnicas');
      const snapshot = await getDocs(colRef);
      const data = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
      setVisitas(data);
    };

    fetchData();
  }, []);

  return visitas;
};

// src/components/maps/MapVisitasTecnicas.tsx

import { useEffect, useState } from "react";
import { MapWrapper } from "../ui/MapWrapper";
import { Marker, Popup } from "react-leaflet";
import { getVisitasTecnicas } from "@/services/firebase/pescaService";
import { VisitaTecnica } from "@/types/visitaTecnica";
import { MediaViewer } from "../midias/MediaViewer";

export default function MapVisitasTecnicas() {
  const [visitas, setVisitas] = useState<VisitaTecnica[]>([]);

  useEffect(() => {
    async function fetchData() {
      const data = await getVisitasTecnicas();
      setVisitas(data);
    }
    fetchData();
  }, []);

  return (
    <MapWrapper>
      {visitas.map((visita, index) => (
        <Marker
          key={index}
          position={[visita.coordenadas.lat, visita.coordenadas.lng]}
        >
          <Popup>
            <div className="text-sm">
              <p><strong>Técnico:</strong> {visita.tecnico}</p>
              <p><strong>Data:</strong> {visita.data}</p>
              <p><strong>Localidade:</strong> {visita.localidade}</p>
              <p><strong>Observações:</strong> {visita.observacoes}</p>
              {visita.midias && visita.midias.length > 0 && (
                <MediaViewer midias={visita.midias} />
              )}
            </div>
          </Popup>
        </Marker>
      ))}
    </MapWrapper>
  );
}

✅ ETAPA 2: Serviço para buscar visitas no Firebase

// src/services/firebase/pescaService.ts

import { db } from "@/lib/firebase";
import { collection, getDocs } from "firebase/firestore";
import { VisitaTecnica } from "@/types/visitaTecnica";

export async function getVisitasTecnicas(): Promise<VisitaTecnica[]> {
  const colRef = collection(db, "pesca", "admin", "visitasTecnicas");

  return snapshot.docs.map((doc) => doc.data() as VisitaTecnica);
}

🧮 Etapas do cálculo:

Calcular número de dias entre dataInicio e dataFim

Definir o tipo de máquina e consumo médio (L/h)

Multiplicar e retornar o total estimado

function calcularEstimativaCombustivel(viveiro: Viveiro): number {
  const { dataInicio, dataFim, maquina = 'retroescavadeira' } = viveiro;

  const horasPorDia = 5; // padrão
  const consumoPorHora = {
    retroescavadeira: 12,
    trator: 10,
    'pá carregadeira': 14,
  };

  const inicio = new Date(dataInicio);
  const fim = new Date(dataFim);
  const dias = Math.ceil((fim.getTime() - inicio.getTime()) / (1000 * 60 * 60 * 24)) || 1;

  const consumoHora = consumoPorHora[maquina];
  const estimativaLitros = dias * horasPorDia * consumoHora;

  return estimativaLitros;
}
